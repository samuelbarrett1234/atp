40 class ATP_LOGIC_API IKnowledgeKernel
41 {
42 public:
43  virtual ~IKnowledgeKernel() = default;
44 
45  // If two knowledge kernels agree on their code, then they agree
46  // on the return results of all the functions below. (This code
47  // is computed by hashing all of the available syntax and axioms
48  // etc.)
49  virtual size_t get_integrity_code() const = 0;
50 
51  // Precondition: valid(p_stmts)
52  // Postcondition: for each statement in the input array, we
53  // return an array of statements which follow in one step
54  // from it.
55  virtual std::vector<StatementArrayPtr> succs(
56  StatementArrayPtr p_stmts) const = 0;
57 
58  // Postcondition: for each statement, check whether it is valid
59  // in terms of the logic, axioms and definitions in this
60  // knowledge kernel. Statements are only allowed to be used if
61  // they are valid. Return true iff all statements in the array
62  // are valid.
63  // Note that a necessary condition for validity is that they
64  // must be using the same language as that which created this
65  // knowledge kernel.
66  virtual bool valid(
67  StatementArrayPtr p_stmts) const = 0;
68 
69  // Precondition: p_premise->size() == p_concl.size() and
70  // valid(p_premise) and valid(p_concl)
71  // Postcondition: for each premise and conclusion, returns
72  // true (in the output array) if the conclusion follows from
73  // the premise in a single step.
74  virtual std::vector<bool> follows(
75  StatementArrayPtr p_premise,
76  StatementArrayPtr p_concl) const = 0;
77 };
